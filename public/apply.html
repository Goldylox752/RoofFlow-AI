import { createClient } from "@supabase/supabase-js"; const supabase = createClient( process.env.SUPABASE_URL, process.env.SUPABASE_SERVICE_KEY ); export default async function handler(req, res) { if (req.method !== "POST") { return res.status(405).json({ error: "Method not allowed" }); } try { const body = req.body || {}; const { name, phone, email, city, monthly_jobs, lead_spend, } = body; // 🔒 Basic validation if (!name || !phone || !email) { return res.status(400).json({ error: "Missing required fields" }); } // 🧠 Qualification rules (centralized) const isLowVolume = monthly_jobs === "0-5"; const noAdSpend = lead_spend === "$0"; const qualified = !(isLowVolume || noAdSpend); // 💾 Insert into Supabase const { error } = await supabase.from("applications").insert({ name, phone, email, city, monthly_jobs, lead_spend, qualified, created_at: new Date().toISOString(), }); if (error) { console.error("Supabase insert error:", error); return res.status(500).json({ error: "Failed to save application", }); } return res.status(200).json({ success: true, qualified, }); } catch (err) { console.error("Server error:", err); return res.status(500).json({ error: "Internal server error", }); } }
